    <script type="text/javascript">
        // วางลิงก์โมเดล Teachable Machine ของคุณในเครื่องหมายคำพูดด้านล่างนี้ครับ
        const URL = "https://teachablemachine.withgoogle.com/models/bPg16Yfub/"; 

        let model, maxPredictions;
        let videoElement = document.getElementById("webcam-video");
        let countdownInterval;
        const INTERVAL_SECONDS = 180; // ลูปเวลา 3 นาที (180 วินาที)
        let timeRemaining = INTERVAL_SECONDS;
        let logCounter = 0;
        let isSystemRunning = false;

        // ฟังก์ชันหลักกดเปิดระบบ
        async function initSystem() {
            document.getElementById("error-alert").classList.add("hidden");
            const btnStart = document.getElementById("btn-start");
            btnStart.innerHTML = `<i class="fa-solid fa-circle-notch animate-spin"></i> กำลังเรียกเปิดกล้องและเชื่อมต่อ AI...`;
            btnStart.disabled = true;

            try {
                // 1. คำสั่งบังคับเรียกเปิดสิทธิ์กล้องฮาร์ดแวร์แบบ Native API การันตีการเปิดกล้องสำเร็จ
                const stream = await navigator.mediaDevices.getUserMedia({ 
                    video: { width: 400, height: 400, facingMode: "user" }, 
                    audio: false 
                });
                
                // นำวิดีโอสดผูกเข้ากับแท็ก Video บน HTML เพื่อให้แสดงผลลื่นไหล
                videoElement.srcObject = stream;
                videoElement.classList.remove("hidden");
                document.getElementById("webcam-placeholder").classList.add("hidden");

                // 2. โหลดไฟล์โครงข่ายประสาทเทียม AI
                model = await tmImage.load(URL + "model.json", URL + "metadata.json");
                maxPredictions = model.getTotalClasses();

                // 3. จัดแจงหน้าต่างแสดงผลแท่ง Progress Bar คะแนนเปอร์เซ็นต์ความแม่นยำ
                const labelContainer = document.getElementById("label-container");
                labelContainer.innerHTML = "";
                for (let i = 0; i < maxPredictions; i++) {
                    const classBlock = document.createElement("div");
                    classBlock.className = "space-y-1";
                    classBlock.innerHTML = `
                        <div class="flex justify-between text-xs font-semibold">
                            <span class="text-slate-300 class-name-lbl">คำนวณชื่อคลาส...</span>
                            <span class="text-indigo-400 font-mono class-prob-lbl">0.00%</span>
                        </div>
                        <div class="w-full bg-slate-950 rounded-full h-2.5 overflow-hidden border border-slate-800">
                            <div class="bg-gradient-to-r from-indigo-500 to-cyan-400 h-2.5 rounded-full transition-all duration-300 class-bar" style="width: 0%"></div>
                        </div>
                    `;
                    labelContainer.appendChild(classBlock);
                }

                // ปรับแต่งการแสดงผล UI เมื่อเปิดกล้องติดสมบูรณ์
                isSystemRunning = true;
                btnStart.innerHTML = `<i class="fa-solid fa-video"></i> กล้องเปิดใช้งานอยู่`;
                
                document.getElementById("system-badge").className = "text-xs bg-emerald-500/20 text-emerald-400 px-2.5 py-0.5 rounded-full font-normal";
                document.getElementById("system-badge").innerText = "Active";
                document.getElementById("status-badge").className = "bg-emerald-500/10 text-emerald-400 border border-emerald-500/20 text-xs px-3 py-1 rounded-full font-mono";
                document.getElementById("status-badge").innerText = "Monitoring";
                document.getElementById("countdown-timer").className = "text-xl font-mono font-bold text-cyan-400";

                const btnPredictNow = document.getElementById("btn-predict-now");
                btnPredictNow.disabled = false;
                btnPredictNow.className = "bg-slate-800 hover:bg-indigo-600 text-white p-2.5 rounded-xl transition-all duration-200 cursor-pointer";

                // ยิงวิเคราะห์ครั้งแรก และเริ่มระเบิดเวลาถอยหลัง 3 นาที
                await predictImage();
                startCountdownTimer();

            } catch (err) {
                console.error("Camera Access Interrupted: ", err);
                document.getElementById("error-alert").classList.remove("hidden");
                btnStart.innerHTML = `<i class="fa-solid fa-triangle-exclamation text-rose-500"></i> เปิดใช้งานไม่สำเร็จ`;
                btnStart.disabled = false;
            }
        }

        // ตัวคุมวงรอบเวลานับถอยหลัง 3 นาที
        function startCountdownTimer() {
            clearInterval(countdownInterval);
            timeRemaining = INTERVAL_SECONDS;

            countdownInterval = setInterval(async () => {
                timeRemaining--;

                let minutes = Math.floor(timeRemaining / 60);
                let seconds = timeRemaining % 60;
                minutes = minutes < 10 ? '0' + minutes : minutes;
                seconds = seconds < 10 ? '0' + seconds : seconds;

                document.getElementById("countdown-timer").innerText = `${minutes}:${seconds}`;

                // เมื่อนาฬิกาตีศูนย์ (ครบ 3 นาที)
                if (timeRemaining <= 0) {
                    await predictImage(); 
                    timeRemaining = INTERVAL_SECONDS; // รีเซ็ตรอบเวลากลับไปเริ่มต้นใหม่
                }
            }, 1000);
        }

        // ปุ่มบังคับทำนายทันที (Manual Bypass)
        async function triggerManualPredict() {
            if (!isSystemRunning) return;
            await predictImage(); 
            startCountdownTimer(); // บังคับรีเวลากลับไปเริ่มต้น 3 นาทีใหม่ทันที   
        }

        // ฟังก์ชันสั่งให้ปัญญาประดิษฐ์สแนปช็อตภาพจากวิดีโอไปจำแนกประเภทคลาส
        async function predictImage() {
            if (!model || !isSystemRunning) return;

            // นำส่ง Frame ปัจจุบันของ Tag วิดีโอเข้าไปที่โมเดลของ Teachable Machine
            const prediction = await model.predict(videoElement);
            let topClass = "";
            let topProbability = -1;
            const now = new Date();
            const timeString = now.toTimeString().split(' ')[0];

            const labelContainer = document.getElementById("label-container");
            for (let i = 0; i < maxPredictions; i++) {
                const className = prediction[i].className;
                const probability = prediction[i].probability;
                const percentString = (probability * 100).toFixed(2) + "%";

                const classBlock = labelContainer.childNodes[i];
                if(classBlock && classBlock.querySelector(".class-name-lbl")) {
                    classBlock.querySelector(".class-name-lbl").innerText = className;
                    classBlock.querySelector(".class-prob-lbl").innerText = percentString;
                    classBlock.querySelector(".class-bar").style.width = percentString;
                }

                // หาค่าเป้าหมายที่มีร้อยละความแม่นยำสูงที่สุดประจำรอบนั้นๆ
                if (probability > topProbability) {
                    topProbability = probability;
                    topClass = className;
                }
            }

            // บันทึกสถิติเวลาและประวัติลงในตาราง Logs
            document.getElementById("last-update-time").innerText = timeString;
            insertNewRowLog(timeString, topClass, (topProbability * 100).toFixed(2));
        }

        // บันทึกแถวข้อมูลลงในตารางประวัติเบื้องล่าง
        function insertNewRowLog(time, className, score) {
            const tableBody = document.getElementById('log-table-body');
            const emptyRow = document.getElementById('empty-row');
            
            if (emptyRow) { emptyRow.remove(); }

            logCounter++;
            document.getElementById('log-count').innerText = `${logCounter} Logs`;

            const newRow = document.createElement('tr');
            newRow.className = "hover:bg-slate-800/30 transition-colors border-b border-slate-800/40";
            newRow.innerHTML = `
                <td class="px-6 py-3.5 font-mono text-xs text-slate-500">${logCounter}</td>
                <td class="px-6 py-3.5 font-mono text-slate-300">${time}</td>
                <td class="px-6 py-3.5 font-semibold text-indigo-300">
                    ${className} <span class="text-xs font-mono font-normal text-cyan-400 ml-1">(${score}%)</span>
                </td>
                <td class="px-6 py-3.5 text-right text-xs"><span class="text-emerald-400 bg-emerald-500/10 px-2 py-0.5 rounded-md"><i class="fa-solid fa-circle-check mr-1"></i>Saved</span></td>
            `;
            // แทรกลงด้านบนสุดของตาราง เพื่อให้อ่านประวัติล่าสุดได้สะดวกที่สุด
            tableBody.insertBefore(newRow, tableBody.firstChild);
        }
    </script>
