    <script src="https://cdn.jsdelivr.net/npm/@teachablemachine/image@latest/dist/teachablemachine-image.min.js"></script>
    
    <script type="text/javascript">
        // ลิงก์ไปยังโฟลเดอร์โมเดลของคุณ (เช่น ลิงก์ที่ได้จาก Teachable Machine online หรือโฟลเดอร์ภายใน)
        const URL = "./my_model/";

        let model, webcam, labelContainer, maxPredictions;
        let countdownInterval;
        const INTERVAL_SECONDS = 180; // ตั้งค่า 3 นาที (180 วินาที)
        let timeRemaining = INTERVAL_SECONDS;
        let logCounter = 0;
        let isSystemRunning = false;

        // ฟังก์ชันเริ่มต้นโมเดลและเว็บแคม (เรียกเมื่อกดปุ่ม)
        async function init() {
            document.getElementById("btn-start").innerHTML = `<i class="fa-solid fa-spinner animate-spin"></i> กำลังโหลดโมเดล...`;
            document.getElementById("btn-start").disabled = true;

            const modelURL = URL + "model.json";
            const metadataURL = URL + "metadata.json";

            try {
                // โหลดโมเดล
                model = await tmImage.load(modelURL, metadataURL);
                maxPredictions = model.getTotalClasses();

                // ตั้งค่ากล้อง Webcam
                const flip = true; 
                webcam = new tmImage.Webcam(400, 400, flip); // ขยายขนาดพิกเซลให้ชัดขึ้นเพื่อความสวยงาม
                await webcam.setup(); 
                await webcam.play();
                
                // ซ่อนหน้าต่างจองที่และเปิดสตรีมกล้อง
                document.getElementById("webcam-placeholder").classList.add("hidden");
                document.getElementById("webcam-container").appendChild(webcam.canvas);
                
                // เคลียร์พื้นที่แสดงผล Label เพื่อเตรียมรับ Progress Bar
                labelContainer = document.getElementById("label-container");
                labelContainer.innerHTML = "";
                for (let i = 0; i < maxPredictions; i++) {
                    // สร้างบล็อก UI สำหรับแต่ละคลาส
                    const classBlock = document.createElement("div");
                    classBlock.className = "space-y-1";
                    classBlock.innerHTML = `
                        <div class="flex justify-between text-xs font-semibold">
                            <span class="text-slate-300 class-name-lbl">Loading Class...</span>
                            <span class="text-indigo-400 font-mono class-prob-lbl">0.00%</span>
                        </div>
                        <div class="w-full bg-slate-950 rounded-full h-2.5 overflow-hidden border border-slate-800">
                            <div class="bg-gradient-to-r from-indigo-500 to-cyan-400 h-2.5 rounded-full transition-all duration-500 class-bar" style="width: 0%"></div>
                        </div>
                    `;
                    labelContainer.appendChild(classBlock);
                }

                // สั่งรันลูปสำหรับการอัปเดตเฟรมภาพเคลื่อนไหวบนหน้าจอ
                window.requestAnimationFrame(webcamLoop);

                // เปลี่ยนแปลงสถานะปุ่มและระบบคอนโทรล
                isSystemRunning = true;
                document.getElementById("btn-start").innerHTML = `<i class="fa-solid fa-check-circle text-emerald-400"></i> AI ทำงานอยู่`;
                document.getElementById("status-badge").className = "bg-emerald-500/10 text-emerald-400 border border-emerald-500/20 text-xs px-3 py-1 rounded-full font-mono";
                document.getElementById("status-badge").innerText = "Monitoring";
                
                // ปลดล็อกปุ่มเร่งทำนายผล (Manual Predict)
                const btnPredictNow = document.getElementById("btn-predict-now");
                btnPredictNow.disabled = false;
                btnPredictNow.className = "bg-slate-800 hover:bg-indigo-600 text-white p-2.5 rounded-xl transition-all duration-200 cursor-pointer group";

                // ทำนายทันทีรอบแรก และเริ่มจับเวลานับถอยหลัง 3 นาที
                await predict();
                startTimer();

            } catch (error) {
                console.error(error);
                alert("ไม่สามารถโหลดโมเดลหรือเข้าถึงกล้องได้ กรุณาตรวจสอบสิทธิ์การเข้าถึงกล้องหรือตำแหน่งของโฟลเดอร์โมเดล");
                document.getElementById("btn-start").innerHTML = `<i class="fa-solid fa-triangle-exclamation text-rose-500"></i> เกิดข้อผิดพลาด`;
                document.getElementById("btn-start").disabled = false;
            }
        }

        // ลูปในการดึงภาพจากกล้องขึ้นมาแสดงบนหน้าจอ (แต่ไม่ได้เรียก predict ทุกเฟรมเพื่อประหยัดทรัพยากร)
        async function webcamLoop() {
            if (isSystemRunning) {
                webcam.update(); // อัปเดตเฟรมกล้องให้เคลื่อนไหวลื่นไหล
                window.requestAnimationFrame(webcamLoop);
            }
        }

        // ฟังก์ชันระบบจับเวลานับถอยหลัง 3 นาที
        function startTimer() {
            clearInterval(countdownInterval);
            timeRemaining = INTERVAL_SECONDS;

            countdownInterval = setInterval(async () => {
                timeRemaining--;

                let minutes = Math.floor(timeRemaining / 60);
                let seconds = timeRemaining % 60;

                minutes = minutes < 10 ? '0' + minutes : minutes;
                seconds = seconds < 10 ? '0' + seconds : seconds;

                document.getElementById("countdown-timer").innerText = `${minutes}:${seconds}`;

                // เมื่อเวลาครบ 3 นาที (เหลือ 0)
                if (timeRemaining <= 0) {
                    await predict(); // เรียกทำนายผลอัปเดตหน้าจอและลงตาราง Log
                    timeRemaining = INTERVAL_SECONDS; // รีเซ็ตตัวนับกลับไป 3 นาทีใหม่
                }
            }, 1000);
        }

        // ฟังก์ชันควบคุมปุ่มเร่งการทำนายผลแบบ Manual
        async function triggerManualPredict() {
            if (!isSystemRunning) return;
            const icon = document.getElementById("refresh-icon");
            icon.className = "fa-solid fa-bolt animate-bounce text-yellow-400";
            
            await predict(); // สั่งทำนายผลทันที
            startTimer();    // รีเซ็ตการนับเวลา 3 นาทีใหม่ของรอบนั้นๆ
            
            setTimeout(() => { icon.className = "fa-solid fa-bolt"; }, 600);
        }

        // ประมวลผลภาพจากกล้องเข้าสู่ตัวโมเดล AI (จะทำงานทุก 3 นาที หรือเมื่อกดปุ่มสายฟ้า)
        async function predict() {
            if (!model || !webcam) return;

            // นำภาพเฟรมปัจจุบันไปให้โมเดลทำนาย
            const prediction = await model.predict(webcam.canvas);
            
            let topClass = "";
            let topProbability = -1;
            const now = new Date();
            const timeString = now.toTimeString().split(' ')[0];

            // วนลูปเพื่อนำค่าผลลัพธ์ไปแสดงผลผ่าน UI Progress bar
            for (let i = 0; i < maxPredictions; i++) {
                const className = prediction[i].className;
                const probability = prediction[i].probability;
                const percentString = (probability * 100).toFixed(2) + "%";

                // ค้นหา Element ในบล็อกที่สร้างไว้เพื่อยัดข้อมูลลงไป
                const classBlock = labelContainer.childNodes[i];
                classBlock.querySelector(".class-name-lbl").innerText = className;
                classBlock.querySelector(".class-prob-lbl").innerText = percentString;
                classBlock.querySelector(".class-bar").style.width = percentString;

                // ตรวจหาคลาสที่มีคะแนนทำนายสูงที่สุดในรอบนี้เพื่อเอาไปลงตารางบันทึกประวัติ
                if (probability > topProbability) {
                    topProbability = probability;
                    topClass = className;
                }
            }

            // แสดงเวลาที่ระบบทำนายผลล่าสุดบนการ์ด
            document.getElementById("last-update-time").innerText = timeString;

            // เพิ่มข้อมูลรอบการทำนายลงในตารางประวัติ (Log) ด้านล่าง
            addNewLog(timeString, topClass, (topProbability * 100).toFixed(2));
        }

        // ฟังก์ชันบันทึกแถวข้อมูลลงตาราง Log ด้านล่าง
        function addNewLog(time, className, score) {
            const tableBody = document.getElementById('log-table-body');
            const emptyRow = document.getElementById('empty-row');
            
            if (emptyRow) { emptyRow.remove(); } // ลบแถวว่างออกเมื่อมีข้อมูลเข้า

            logCounter++;
            document.getElementById('log-count').innerText = `${logCounter} Logs`;

            const newRow = document.createElement('tr');
            newRow.className = "hover:bg-slate-800/30 transition-colors border-b border-slate-800/40 animate-fadeIn";
            newRow.innerHTML = `
                <td class="px-6 py-3.5 font-mono text-xs text-slate-500">${logCounter}</td>
                <td class="px-6 py-3.5 font-mono text-slate-300">${time}</td>
                <td class="px-6 py-3.5 font-semibold text-indigo-300">
                    ${className} <span class="text-xs font-mono font-normal text-cyan-400 ml-1">(${score}%)</span>
                </td>
                <td class="px-6 py-3.5 text-right text-xs"><span class="text-emerald-400 bg-emerald-500/10 px-2 py-0.5 rounded-md"><i class="fa-solid fa-bolt mr-1"></i>Predicted</span></td>
            `;

            // เพิ่มแถวใหม่ไว้บนสุด เพื่อให้เห็นประวัติล่าสุดเสมอ
            tableBody.insertBefore(newRow, tableBody.firstChild);
        }
    </script>
